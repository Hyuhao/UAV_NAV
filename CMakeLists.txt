cmake_minimum_required(VERSION 2.8.3)
project(uav_nav)
set(CMAKE_CXX_FLAGS "-std=c++11 ${CMAKE_CXX_FLAGS}")

find_package(catkin REQUIRED COMPONENTS
  cv_bridge
  geometry_msgs
  nav_msgs
  roscpp
  rospy
  std_msgs
  sensor_msgs
  dji_sdk
  roslaunch
  message_generation
)
find_package(DJIOSDK REQUIRED)
find_package(OpenCV REQUIRED)


roslaunch_add_file_check(launch)

add_service_files(
  FILES
  VFHLookUpTables.srv
)

generate_messages(
  DEPENDENCIES
)

catkin_package()

link_directories(lib/x64)

include_directories(
  include
  ${DJIOSDK_INCLUDE_DIRS}
  ${OpenCV_INCLUDE_DIRS}
  ${catkin_INCLUDE_DIRS}
)

add_executable(sensor_feedback
  src/sensor_feedback.cc
  src/DJI_utility.cc)
target_link_libraries(sensor_feedback
  ${catkin_LIBRARIES}
  ${OpenCV_LIBS}
  ${DJIOSDK_LIBRARIES}
  DJI_guidance
  usb-1.0
  yaml-cpp
)


add_executable(depth_generation
  src/depth_generation.cc
  src/supporting_alg.cc
)
target_link_libraries(depth_generation
  ${OpenCV_LIBS}
  ${catkin_LIBRARIES}
  DJI_guidance
  usb-1.0
  yaml-cpp
)

add_executable(drone_control src/drone_control.cc)
target_link_libraries(drone_control
  ${catkin_LIBRARIES}
  ${DJIOSDK_LIBRARIES})

add_executable(test_vel_cmd src/test_vel_cmd.cc)
target_link_libraries(test_vel_cmd
  ${catkin_LIBRARIES}
  ${DJIOSDK_LIBRARIES})

add_executable(vfh src/vfh.cc)
target_link_libraries(vfh
  ${catkin_LIBRARIES}
  ${OpenCV_LIBS})

add_executable(vfh_server src/vfh_services.cc)
target_link_libraries(vfh_server ${catkin_LIBRARIES})
add_dependencies(vfh_server uav_nav_gencpp)
